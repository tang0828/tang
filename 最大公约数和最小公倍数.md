#include<stdio.h>

void gcd(long int a,long int b){

    if(a<0||b<0){
    
        printf("Input Error");
        
    }else{
    
     long long int A,B;
     
     long long int c,d;
     
    A=a;
    
    B=b;
    
    if(B){
    
        while((A%=B)&&(B%=A));//辗转相除法
        
        c=A+B;//最大公约数
        
        d=a*b/c;//最小公倍数
        
    }
    printf("%d %ld",c,d);
    
}

}
int main(void)
{  
    
    long int a,b;
    scanf("%ld %ld",&a,&b);
    gcd(a,b);
    
    return 0;
}
